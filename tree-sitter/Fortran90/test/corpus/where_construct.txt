PROGRAM WHERE_EXAMPLE
  REAL :: A(10), B(10), C(10)
  
  WHERE (A > 0.0)
    B = SQRT(A)
    C = 2.0 * A
  ELSEWHERE (A < 0.0)
    B = 0.0
    C = -A
  ELSEWHERE
    B = 1.0
    C = 1.0
  END WHERE
END PROGRAM
