PROGRAM DYNAMIC_ARRAYS
  INTEGER :: N = 100
  REAL, ALLOCATABLE :: X(:), Y(:)
  
  ALLOCATE(X(N), Y(N))
  
  X = (/ (I, I=1,N) /)
  Y = 2.0 * X
  
  DEALLOCATE(X, Y)
END PROGRAM
